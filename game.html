let randomNumber = Math.floor(Math.random() * 10) + 1; let attempts = 0; function checkGuess() { const input = document.getElementById("guess"); const result = document.getElementById("result"); let userGuess = parseInt(input.value); attempts++; if (isNaN(userGuess)) { result.innerHTML = "⚠️ Masukkan angka dulu!"; return; } if (userGuess === randomNumber) { result.innerHTML = `🎉 Benar! Kamu menebak dalam ${attempts} kali!`; // reset game randomNumber = Math.floor(Math.random() * 10) + 1; attempts = 0; input.value = ""; } else if (userGuess < randomNumber) { result.innerHTML = "📉 Terlalu kecil!"; } else { result.innerHTML = "📈 Terlalu besar!"; } } // biar bisa tekan ENTER juga document.getElementById("guess").addEventListener("keydown", function(event) { if (event.key === "Enter") { checkGuess(); } });
